// ─── BLOB MATCHING ───────────────────────────

function matchBlobs(prev, centroids) {

  const out={}, used=new Set(), MAX=80;

  for (const [cx,cy] of centroids) {

    let bid=null, bd=MAX;

    for (const [tid,tr] of Object.entries(prev)) {

      if (used.has(tid)||!tr.length) continue;

      const l=tr[tr.length-1];

      const d=Math.hypot(cx-l.x, cy-l.y);

      if (d<bd){bd=d;bid=tid;}

    }

    if (bid!==null) {

      used.add(bid);

      const t=prev[bid]; t.push({x:cx,y:cy});

      if(t.length>TRAIL_LEN) t.shift();

      out[bid]=t;

    } else {

      out[nextId]=[{x:cx,y:cy}];

      nextId++;

    }

  }

  return out;

}


// ─── MOTION DETECTION ────────────────────────

function getMotionMask(curr, prev, W, H, thr) {

  const m = new Uint8Array(W*H);

  for (let i=0;i<W*H;i++) {

    const p=i*4;

    const lc=0.299*curr[p]+0.587*curr[p+1]+0.114*curr[p+2];

    const lp=0.299*prev[p]+0.587*prev[p+1]+0.114*prev[p+2];

    m[i]=Math.abs(lc-lp)>thr?255:0;

  }

  return m;

}

function dilateMask(m,W,H,r=4) {

  const o=new Uint8Array(W*H);

  for(let y=0;y<H;y++) for(let x=0;x<W;x++) {

    let f=false;

    outer: for(let dy=-r;dy<=r;dy++) for(let dx=-r;dx<=r;dx++){

      const nx=x+dx,ny=y+dy;

      if(nx<0||nx>=W||ny<0||ny>=H) continue;

      if(m[ny*W+nx]===255){f=true;break outer;}

    }

    o[y*W+x]=f?255:0;

  }

  return o;

}

function findBlobs(m,W,H,minA,maxA) {

  const vis=new Uint8Array(W*H), blobs=[];

  for(let sy=0;sy<H;sy++) for(let sx=0;sx<W;sx++){

    const si=sy*W+sx;

    if(m[si]!==255||vis[si]) continue;

    const q=[si]; vis[si]=1;

    let mnx=sx,mxx=sx,mny=sy,mxy=sy,px=0;

    while(q.length){

      const i=q.shift(), qx=i%W, qy=(i-qx)/W;

      px++; mnx=Math.min(mnx,qx); mxx=Math.max(mxx,qx);

      mny=Math.min(mny,qy); mxy=Math.max(mxy,qy);

      for(const n of [i-1,i+1,i-W,i+W]){

        if(n<0||n>=W*H||vis[n]||m[n]!==255) continue;

        vis[n]=1; q.push(n);

      }

    }

    const bw=mxx-mnx+1,bh=mxy-mny+1;

    if(px<minA||px>maxA) continue;

    blobs.push({x:mnx,y:mny,w:bw,h:bh,

                cx:Math.round(mnx+bw/2),

                cy:Math.round(mny+bh/2),area:px});

  }

  return blobs;

}


// ─── CONTRAST COLOR ──────────────────────────

function contrastColor(d,x,y,w,h,W,H) {

  let s=0,c=0;

  for(let iy=Math.max(0,y-8);iy<Math.min(H,y+4);iy++)

    for(let ix=Math.max(0,x);ix<Math.min(W,x+w);ix++){

      const p=(iy*W+ix)*4;

      s+=0.299*d[p]+0.587*d[p+1]+0.114*d[p+2];c++;

    }

  return c&&s/c<128?'#00f5ff':'#3c14dc';

}


// ─── CONVEX HULL ─────────────────────────────

function cross(O,A,B){

  return (A[0]-O[0])*(B[1]-O[1])-(A[1]-O[1])*(B[0]-O[0]);

}

function convexHull(pts){

  const n=pts.length; if(n<3) return pts;

  const p=[...pts].sort((a,b)=>a[0]-b[0]||a[1]-b[1]);

  const lo=[],hi=[];

  for(const pt of p){

    while(lo.length>=2&&cross(lo[lo.length-2],lo[lo.length-1],pt)<=0) lo.pop();

    lo.push(pt);

  }

  for(let i=p.length-1;i>=0;i--){

    const pt=p[i];

    while(hi.length>=2&&cross(hi[hi.length-2],hi[hi.length-1],pt)<=0) hi.pop();

    hi.push(pt);

  }

  return lo.slice(0,-1).concat(hi.slice(0,-1));

}

function drawHull(c2,blobs){

  if(blobs.length<2) return;

  const pts=blobs.map(b=>[b.cx,b.cy]);

  const gcx=pts.reduce((s,p)=>s+p[0],0)/pts.length;

  const gcy=pts.reduce((s,p)=>s+p[1],0)/pts.length;

  c2.strokeStyle='#ff8844'; c2.fillStyle='#ff8844'; c2.lineWidth=1;

  for(const b of blobs){

    c2.beginPath(); c2.moveTo(b.cx,b.cy); c2.lineTo(gcx,gcy); c2.stroke();

  }

  c2.beginPath(); c2.arc(gcx,gcy,5,0,Math.PI*2); c2.fill();

  c2.font='11px -apple-system,sans-serif';

  c2.fillText('G',gcx+7,gcy+4);

  if(blobs.length>=3){

    const h=convexHull(pts);

    c2.beginPath(); c2.moveTo(h[0][0],h[0][1]);

    for(let i=1;i<h.length;i++) c2.lineTo(h[i][0],h[i][1]);

    c2.closePath(); c2.stroke();

  }

}


// ─── VORONOI ─────────────────────────────────

// Uses a fast jump-flood approach on a small canvas then scales up

function drawVoronoi(c2,blobs,W,H){

  if(blobs.length<2) return;

  const S=4, sw=Math.ceil(W/S), sh=Math.ceil(H/S);

  const vc=document.createElement('canvas');

  vc.width=sw; vc.height=sh;

  const vx=vc.getContext('2d');

  // Colour palette per blob

  const cols=['#bb88ff22','#00cfff22','#ff888822','#00ff8822',

              '#ffaa0022','#ff88cc22','#88ffcc22','#4f8ef722'];

  // For each pixel find nearest blob

  const id=new ImageData(sw,sh);

  for(let y=0;y<sh;y++) for(let x=0;x<sw;x++){

    let bd=Infinity,bc=0;

    for(let i=0;i<blobs.length;i++){

      const d=Math.hypot(x*S-blobs[i].cx, y*S-blobs[i].cy);

      if(d<bd){bd=d;bc=i;}

    }

    const col=hexToRgb(cols[bc%cols.length]);

    const p=(y*sw+x)*4;

    id.data[p]=col[0]; id.data[p+1]=col[1];

    id.data[p+2]=col[2]; id.data[p+3]=col[3];

  }

  vx.putImageData(id,0,0);

  c2.save();

  c2.globalAlpha=0.55;

  c2.imageSmoothingEnabled=false;

  c2.drawImage(vc,0,0,W,H);

  c2.restore();

  // Draw cell borders

  c2.strokeStyle='rgba(187,136,255,0.4)';

  c2.lineWidth=0.5;

  // Approximate borders by edge detection on nearest-blob grid

  const grid=new Int16Array(sw*sh);

  for(let y=0;y<sh;y++) for(let x=0;x<sw;x++){

    let bc=0,bd=Infinity;

    for(let i=0;i<blobs.length;i++){

      const d=Math.hypot(x*S-blobs[i].cx, y*S-blobs[i].cy);

      if(d<bd){bd=d;bc=i;}

    }

    grid[y*sw+x]=bc;

  }

  c2.beginPath();

  for(let y=0;y<sh-1;y++) for(let x=0;x<sw-1;x++){

    const c=grid[y*sw+x];

    if(grid[y*sw+x+1]!==c||grid[(y+1)*sw+x]!==c){

      c2.moveTo(x*S,y*S); c2.lineTo((x+1)*S,y*S);

    }

  }

  c2.stroke();

}

function hexToRgb(hex){

  const r=parseInt(hex.slice(1,3),16)||0;

  const g=parseInt(hex.slice(3,5),16)||0;

  const b=parseInt(hex.slice(5,7),16)||0;

  const a=hex.length>7?parseInt(hex.slice(7,9),16):255;

  return [r,g,b,a];

}


// ─── FORCE FIELD ─────────────────────────────

function drawForceField(c2,blobs,W,H){

  if(!blobs.length) return;

  const step=28;

  c2.strokeStyle='rgba(0,207,255,0.55)';

  c2.fillStyle='rgba(0,207,255,0.55)';

  c2.lineWidth=0.8;

  for(let y=step/2;y<H;y+=step) for(let x=step/2;x<W;x+=step){

    let fx=0,fy=0;

    for(const b of blobs){

      const dx=b.cx-x, dy=b.cy-y;

      const d2=dx*dx+dy*dy+1;

      const f=3000/d2;

      fx+=f*dx/Math.sqrt(d2);

      fy+=f*dy/Math.sqrt(d2);

    }

    const mag=Math.hypot(fx,fy);

    if(mag<0.3) continue;

    const sc=Math.min(12,mag);

    const nx=fx/mag, ny=fy/mag;

    const ex=x+nx*sc, ey=y+ny*sc;

    c2.beginPath(); c2.moveTo(x,y); c2.lineTo(ex,ey); c2.stroke();

    // arrowhead

    const angle=Math.atan2(ey-y,ex-x);

    c2.beginPath();

    c2.moveTo(ex,ey);

    c2.lineTo(ex-5*Math.cos(angle-0.4),ey-5*Math.sin(angle-0.4));

    c2.lineTo(ex-5*Math.cos(angle+0.4),ey-5*Math.sin(angle+0.4));

    c2.closePath(); c2.fill();

  }

}


// ─── LISSAJOUS ───────────────────────────────

function updateLissa(blobs){

  const ids=new Set(blobs.map((_,i)=>{

    const tid=findTid(blobs[i].cx,blobs[i].cy);

    return tid;

  }));

  // Prune old

  for(const k of Object.keys(lissaHistory)){

    if(!ids.has(+k)) delete lissaHistory[k];

  }

}

function drawLissajous(c2,blobs,W,H){

  for(const b of blobs){

    const tid=findTid(b.cx,b.cy);

    if(tid===null) continue;

    if(!lissaHistory[tid]) lissaHistory[tid]={xs:[],ys:[]};

    const lh=lissaHistory[tid];

    lh.xs.push(b.cx); lh.ys.push(b.cy);

    if(lh.xs.length>TRAIL_LEN){ lh.xs.shift(); lh.ys.shift(); }

    if(lh.xs.length<4) continue;

    // Map x motion to cosine, y motion to sine → parametric

    const n=lh.xs.length;

    const cx0=lh.xs.reduce((s,v)=>s+v,0)/n;

    const cy0=lh.ys.reduce((s,v)=>s+v,0)/n;

    const ax=Math.max(...lh.xs)-Math.min(...lh.xs)||1;

    const ay=Math.max(...lh.ys)-Math.min(...lh.ys)||1;

    c2.strokeStyle='rgba(0,255,136,0.7)';

    c2.lineWidth=1;

    c2.beginPath();

    for(let i=0;i<n;i++){

      const t=i/n*Math.PI*2;

      // Lissajous: encode x-displacement as A*cos(t), y-displacement as B*sin(t)

      const px=b.cx + (lh.xs[i]-cx0)/ax * 40 * Math.cos(t*1.5);

      const py=b.cy + (lh.ys[i]-cy0)/ay * 40 * Math.sin(t*2);

      i===0 ? c2.moveTo(px,py) : c2.lineTo(px,py);

    }

    c2.stroke();

  }

}


// ─── DELAUNAY ────────────────────────────────

// Bowyer-Watson algorithm

function delaunay(points){

  if(points.length<3) return [];

  const M=100000;

  const superTri=[[-M,-M],[M*3,-M],[0,M*3]];

  let tris=[[0,1,2]];

  const pts=[...superTri,...points];

  for(let pi=3;pi<pts.length;pi++){

    const px=pts[pi][0], py=pts[pi][1];

    const bad=[];

    for(const t of tris){

      const [ax,ay]=pts[t[0]], [bx,by]=pts[t[1]], [cx2,cy2]=pts[t[2]];

      const D=2*(ax*(by-cy2)+bx*(cy2-ay)+cx2*(ay-by));

      if(Math.abs(D)<1e-10) continue;

      const ux=((ax*ax+ay*ay)*(by-cy2)+(bx*bx+by*by)*(cy2-ay)+(cx2*cx2+cy2*cy2)*(ay-by))/D;

      const uy=((ax*ax+ay*ay)*(cx2-bx)+(bx*bx+by*by)*(ax-cx2)+(cx2*cx2+cy2*cy2)*(bx-ax))/D;

      const r=Math.hypot(ax-ux,ay-uy);

      if(Math.hypot(px-ux,py-uy)<r) bad.push(t);

    }

    const boundary=[];

    for(const t of bad){

      for(let i=0;i<3;i++){

        const e=[t[i],t[(i+1)%3]];

        const shared=bad.some(t2=>t2!==t&&t2.includes(e[0])&&t2.includes(e[1]));

        if(!shared) boundary.push(e);

      }

    }

    for(const t of bad){

      const i=tris.indexOf(t);

      if(i>-1) tris.splice(i,1);

    }

    for(const e of boundary) tris.push([e[0],e[1],pi]);

  }

  return tris.filter(t=>!t.some(i=>i<3))

             .map(t=>t.map(i=>[pts[i][0],pts[i][1]]));

}

function drawDelaunay(c2,blobs){

  if(blobs.length<3) return;

  const pts=blobs.map(b=>[b.cx,b.cy]);

  const tris=delaunay(pts);

  c2.strokeStyle='rgba(255,170,0,0.55)';

  c2.fillStyle='rgba(255,170,0,0.06)';

  c2.lineWidth=0.8;

  for(const t of tris){

    c2.beginPath();

    c2.moveTo(t[0][0],t[0][1]);

    c2.lineTo(t[1][0],t[1][1]);

    c2.lineTo(t[2][0],t[2][1]);

    c2.closePath();

    c2.fill(); c2.stroke();

  }

}


// ─── FOURIER RINGS ───────────────────────────

function updateFourierRings(blobs,prevBlobs){

  // Emit a ring when a blob changes direction significantly

  for(const b of blobs){

    const tid=findTid(b.cx,b.cy);

    if(tid===null) continue;

    const tr=blobTrails[tid];

    if(!tr||tr.length<4) continue;

    const n=tr.length;

    const v1x=tr[n-2].x-tr[n-3].x, v1y=tr[n-2].y-tr[n-3].y;

    const v2x=tr[n-1].x-tr[n-2].x, v2y=tr[n-1].y-tr[n-2].y;

    const dot=v1x*v2x+v1y*v2y;

    const m1=Math.hypot(v1x,v1y), m2=Math.hypot(v2x,v2y);

    if(m1<1||m2<1) continue;

    const angle=Math.acos(Math.max(-1,Math.min(1,dot/(m1*m2))));

    const speed=Math.hypot(v2x,v2y);

    if(angle>0.2&&speed>2){

      fourierRings.push({ x:b.cx, y:b.cy, r:0,

                          maxR: 30+speed*4, age:0 });

    }

  }

  // Age rings

  fourierRings = fourierRings

    .map(r=>({...r, r:r.r+2.5, age:r.age+1}))

    .filter(r=>r.r<r.maxR);

}

function drawFourierRings(c2){

  for(const r of fourierRings){

    const alpha=1-(r.r/r.maxR);

    c2.strokeStyle=`rgba(255,136,204,${alpha*0.7})`;

    c2.lineWidth=0.8;

    c2.beginPath();

    c2.arc(r.x,r.y,r.r,0,Math.PI*2);

    c2.stroke();

    // Harmonic inner ring at half radius

    if(r.r>10){

      c2.strokeStyle=`rgba(255,136,204,${alpha*0.3})`;

      c2.beginPath();

      c2.arc(r.x,r.y,r.r*0.5,0,Math.PI*2);

      c2.stroke();

    }

  }

}


// ─── REACTION DIFFUSION ──────────────────────

function initRD(){

  const w=Math.ceil(cvs.width/RD_SCALE)||80;

  const h=Math.ceil(cvs.height/RD_SCALE)||60;

  if(rdW===w&&rdH===h&&rdA) return;

  rdW=w; rdH=h;

  rdA=new Float32Array(w*h).fill(1);

  rdB=new Float32Array(w*h).fill(0);

}

function seedRD(blobs,W,H){

  if(!rdA) return;

  for(const b of blobs){

    const sx=Math.floor(b.cx/RD_SCALE);

    const sy=Math.floor(b.cy/RD_SCALE);

    for(let dy=-2;dy<=2;dy++) for(let dx=-2;dx<=2;dx++){

      const nx=sx+dx, ny=sy+dy;

      if(nx<0||nx>=rdW||ny<0||ny>=rdH) continue;

      rdB[ny*rdW+nx]=1;

    }

  }

}

function stepRD(steps=3){

  if(!rdA) return;

  // Gray-Scott model: dA/dt = Da*∇²A - A*B² + f*(1-A)

  //                  dB/dt = Db*∇²B + A*B² - (k+f)*B

  const Da=0.16, Db=0.08, f=0.060, k=0.062;

  const nA=new Float32Array(rdA), nB=new Float32Array(rdB);

  const W=rdW, H=rdH;

  for(let s=0;s<steps;s++){

    for(let y=1;y<H-1;y++) for(let x=1;x<W-1;x++){

      const i=y*W+x;

      const lapA=rdA[i-1]+rdA[i+1]+rdA[i-W]+rdA[i+W]-4*rdA[i];

      const lapB=rdB[i-1]+rdB[i+1]+rdB[i-W]+rdB[i+W]-4*rdB[i];

      const a=rdA[i], b=rdB[i];

      const abb=a*b*b;

      nA[i]=Math.max(0,Math.min(1, a+Da*lapA-abb+f*(1-a)));

      nB[i]=Math.max(0,Math.min(1, b+Db*lapB+abb-(k+f)*b));

    }

    rdA.set(nA); rdB.set(nB);

  }

}

function drawRD(c2,W,H){

  if(!rdA||!rdB) return;

  const id=new ImageData(rdW,rdH);

  for(let i=0;i<rdW*rdH;i++){

    const v=Math.max(0,Math.min(255, (rdA[i]-rdB[i])*255));

    const p=i*4;

    // Map to teal/dark palette

    id.data[p]  =Math.round((1-rdA[i])*0+rdB[i]*136);

    id.data[p+1]=Math.round(rdA[i]*80+rdB[i]*255);

    id.data[p+2]=Math.round(rdA[i]*60+rdB[i]*204);

    id.data[p+3]=Math.round(rdB[i]*180);

  }

  const tc=document.createElement('canvas');

  tc.width=rdW; tc.height=rdH;

  tc.getContext('2d').putImageData(id,0,0);

  c2.save();

  c2.globalAlpha=0.75;

  c2.imageSmoothingEnabled=true;

  c2.drawImage(tc,0,0,W,H);

  c2.restore();

}


// ─── HELPERS ─────────────────────────────────

function findTid(cx,cy){

  let best=null,bd=40;

  for(const [tid,tr] of Object.entries(blobTrails)){

    if(!tr.length) continue;

    const l=tr[tr.length-1];

    const d=Math.hypot(cx-l.x,cy-l.y);

    if(d<bd){bd=d;best=+tid;}

  }

  return best;

}


// ─── MAIN LOOP ───────────────────────────────

function loop(){

  animId=requestAnimationFrame(loop);

  if(!vid.videoWidth||vid.readyState<2) return;


  const W=vid.videoWidth, H=vid.videoHeight;

  if(cvs.width!==W||cvs.height!==H){

    cvs.width=W; cvs.height=H;

    offC.width=W; offC.height=H;

    if(ovs.rdiff) initRD();

  }


  offX.drawImage(vid,0,0,W,H);

  const curr=offX.getImageData(0,0,W,H);


  // FPS

  frameCount++;

  const now=performance.now();

  if(now-lastTime>=1000){

    document.getElementById('fps-label').textContent=frameCount+' fps';

    frameCount=0; lastTime=now;

  }


  const thr=Math.max(1,60-sliders.sensitivity);

  let blobs=[];

  if(prevFrame){

    let mask=getMotionMask(curr.data,prevFrame.data,W,H,thr);

    mask=dilateMask(mask,W,H,4);

    blobs=findBlobs(mask,W,H,sliders.minsize,sliders.maxsize);

  }

  if(!prevFrame) prevFrame=curr;

  else {

    for(let i=0;i<curr.data.length;i++)

      prevFrame.data[i]=prevFrame.data[i]*0.7+curr.data[i]*0.3;

  }


  const prevBlobs=Object.values(blobTrails).map(t=>t[t.length-1]).filter(Boolean);

  blobTrails=matchBlobs(blobTrails, blobs.map(b=>[b.cx,b.cy]));


  // RD seed

  if(ovs.rdiff){ if(!rdA) initRD(); seedRD(blobs,W,H); stepRD(3); }


  // Fourier ring update

  if(ovs.fourier) updateFourierRings(blobs,prevBlobs);


  // ── DRAW ──────────────────────────────────

  ctx.clearRect(0,0,W,H);

  ctx.drawImage(vid,0,0,W,H);


  // Reaction diffusion (behind everything)

  if(ovs.rdiff) drawRD(ctx,W,H);


  // Voronoi

  if(ovs.voro&&blobs.length>=2) drawVoronoi(ctx,blobs,W,H);


  // Force field

  if(ovs.force&&blobs.length) drawForceField(ctx,blobs,W,H);


  // Delaunay

  if(ovs.dela&&blobs.length>=3) drawDelaunay(ctx,blobs);


  // Convex hull

  if(ovs.hull&&blobs.length>=2) drawHull(ctx,blobs);


  // Fourier rings

  if(ovs.fourier) drawFourierRings(ctx);


  // Lissajous

  if(ovs.lissa&&blobs.length) drawLissajous(ctx,blobs,W,H);


  // Overlap detection

  const ovSet=new Set();

  for(let i=0;i<blobs.length;i++)

    for(let j=i+1;j<blobs.length;j++){

      const a=blobs[i],b=blobs[j];

      if(a.x<b.x+b.w&&a.x+a.w>b.x&&a.y<b.y+b.h&&a.y+a.h>b.y){

        ovSet.add(i); ovSet.add(j);

      }

    }


  // Blob boxes + labels

  for(let i=0;i<blobs.length;i++){

    const b=blobs[i];

    const isOv=ovSet.has(i);

    const col=isOv?OVERLAP_COL:contrastColor(curr.data,b.x,b.y,b.w,b.h,W,H);


    ctx.strokeStyle=col; ctx.lineWidth=1;

    ctx.strokeRect(b.x,b.y,b.w,b.h);


    // Corner ticks

    ctx.lineWidth=2;

    const t=10;

    for(const [cx2,cy2,dx,dy] of [

      [b.x,b.y,1,1],[b.x+b.w,b.y,-1,1],

      [b.x,b.y+b.h,1,-1],[b.x+b.w,b.y+b.h,-1,-1]

    ]){

      ctx.beginPath();

      ctx.moveTo(cx2,cy2); ctx.lineTo(cx2+dx*t,cy2);

      ctx.moveTo(cx2,cy2); ctx.lineTo(cx2,cy2+dy*t);

      ctx.stroke();

    }


    if(isOv){

      ctx.fillStyle=OVERLAP_COL;

      ctx.font='11px -apple-system,sans-serif';

      ctx.fillText('OVERLAP',b.x,b.y-6);

    }


    const lbl=`#${i+1}  ${b.cx},${b.cy}`;

    let lx=b.x+b.w+8; const ly=b.y+14;

    ctx.font='11px -apple-system,sans-serif';

    const tw=ctx.measureText(lbl).width;

    if(lx+tw>W) lx=b.x-tw-8;

    ctx.fillStyle=col;

    ctx.fillText(lbl,lx,ly);

  }


  document.getElementById('blob-badge').textContent=

    blobs.length+(blobs.length===1?' blob':' blobs');

}


// ─── INIT ────────────────────────────────────

setSource('camera');

loop();
